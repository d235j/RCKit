00013 /// \class Transceiver Transceiver.h <Transceiver.h>
00014 /// \brief Abstract base class for a transceiver device that will send and receive messages
00015 /// on behalf of RCRx class.
00016 ///
00017 /// RXRx class requires to be connected to a Transceiver, and the transceiver is responsible for
00018 /// receiving requests from an RCOIP transmitter and sending replies back to it. 
00019 /// Any transport or media can in principle be supported. RCRx includes some specific implementations
00020 /// for example for WiShield and Ethernet
00021 ///
00022 /// Real transceiver objects must inherit from this subclass and implement the 
00023 /// sendReply() function and call delegate->handleRequest(uint8_t *msg, uint16_t len, uint16_t rssi)
00024 /// whenever an RCOI message is received.
00025 class Transceiver
00026 {   
00027 public:
00028     /// Initialise the object.
00029     /// You are required to call this once during setup().
00030     /// Subclasses must implement this: there is no base implementation.
00031     virtual void init();
00032 
00033     /// Poll the object for activity.
00034     /// You are required to call this frequently in your main loop
00035     /// Subclasses must implement this: there is no base implementation.
00036     virtual void run();
00037 
00038     /// Sets the delegate object.
00039     /// The delegate will get its handleRequest(uint8_t *msg, uint16_t len, uint16_t rssi) 
00040     /// function called whenever an RCOIP message is received.
00041     /// The delegate can use the sendReply() function to send an RCOIP message back to the 
00042     /// sender of the currently received message.
00043     /// \param[in] delegate Pointer to the delegate RXRx object
00044     virtual void   setDelegate(RCRx* delegate);
00045 
00046     /// Passes the received message to the delegate.
00047     /// Call this when the Transceiver has received a message from the RCOIP transmitter.
00048     /// The optional rssi parameter is an indication of the received signal strength, but if your
00049     /// Transceiver has no such measure, you dont need to supply it
00050     /// \param[in] msg Pointer to the RCOP message.
00051     /// \param[in] len Length of the RCOIP message in bytes.
00052     /// \param[in] rssi Receiver Signal Strength as reported by the WiFi receiver (if any)
00053     virtual void receivedRequest(uint8_t* msg, uint16_t len, uint16_t rssi = 0);
00054 
00055     /// Sends an RCOIP reply message to the sender of the current message
00056     /// Subclasses must implement this: there is no base implementation.
00057     /// \param[in] msg Pointer to the RCOP message.
00058     /// \param[in] len Length of the RCOIP message in bytes.
00059     virtual void   sendReply(uint8_t* msg, uint16_t len);
00060 
00061 protected:
