00017 /// \brief Setter class that outputs its value to a MotorController configured on 1 digital and one analog output pins
00018 ///
00019 /// A MotorController is used to drive a motor in forward and reverse directions. It requires 2 Setters as output:
00020 /// one to specify the direction, and one to specify the speed. 
00021 ///
00022 /// Inputs 0 to 127 set direction to 0 (reverse) and speed to 254 to 0 respectively
00023 /// Inputs 128 to 255 set direction to 1 (forward) and speed to 0 to 256 respectively
00024 /// This means than an input of 0 will produce full speed reverse, and input of 127 will produce stop and 
00025 /// 255 will produce full speed forward. Truth table:
00026 /// input   A    B
00027 /// 0       0    254
00028 /// 100     0    54
00029 /// 127     255  0
00030 /// 128     255  2
00031 /// 203     255  146
00032 /// 255     255  256
00033 ///
00034 /// This is compatible with many motor controllers, such as the Arduino Motor Shield (L298N) (SKU:DRI0009) and many others
00035 ///
00036 /// Typically the outputs would be a DigitalSetter and an AnalogSetter to control a motor through a digital (direction) 
00037 /// and an analog (PWM speed) outputs
00038 ///
00039 class MotorControllerSetter : public Setter
00040 {   
00041 public:
00042     /// \param[in] targetA The Setter to use for direction, typically a DigitalSetter.
00043     /// \param[in] targetB The Setter to use for speed, typically an AnalogSetter.
00044     MotorControllerSetter(Setter* targetA, Setter* targetB);
00045 
00046     /// Set or change the output pin
00047     /// \param[in] targetA The Setter to use for output A.
00048     /// \param[in] targetB The Setter to use for output B.
00049     void setTargets(Setter* targetA, Setter* targetB);
00050 
00051     /// Input the value to be used to set the 2 output Setters.
00052     /// Input of 127 produces 255 on A and 0 on B (forward but stopped)
00053     /// Input of 128 to 255 translates to output Setter A to 256 (forward)
00054     /// and output Setter B at 0 to 254 (full speed).
00055     /// Input of 127 to 0 translates to output Setter A to 0 (reverse)
00056     /// and output Setter B at 0 to 254 (full speed).
00057     /// \param[in] value The input value to set.
00058     virtual void input(int value);
00059 
00060     /// Called when the source of input data is lost, and the Setter is required to fail in a safe way.
00061     /// Calls the failsafes of targetA and targetB
00062     virtual void failsafe();
00063 
00064 protected:
00065 
00066 private:
00067     /// The B output Setter. targetA is the setter that is included in the Setter class
